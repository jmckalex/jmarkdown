\begin{definition}[Macro-coarsenings]

A definition whose label sits inside the body, as \label would in LaTeX.
\end{definition}

\begin{theorem}[Main result]
\label{thm-main}
A theorem labelled the same way shares the sequential counter.
\end{theorem}
